int main() {
    int arr2[30];
    int arr[30];

    int p = 1;
    int n = 0;

    for (int i = 0; i < 30; i++) {
        arr2[i] = rand() % 200 - 100;
        arr[i] = arr2[i];
        cout << arr2[i] << " ";
    }

//quick
    n = 0;
    p = 0;
    int u = 29;
    for (int i = 0; i < 600; i++)
        p++;
        if (p == 29) p = 0;
        if (arr2[p] >= arr2[u])
            n++;
        p;

        for (int p = 0; p < 30; p++);
            cout << arr2[p] << " ";
        cout << endl << n << endl;
        return 0;
    };
//cocktail
void cocktail (int arr[], int n) {
    int8_t v = 1;
    int str = 0;
    int end = n - 1;
    while (v) {
        for (int i = str; i < end; ++i) {
            if (arr[i] > arr[i + 1]) {
                int k = arr[i];
                arr[i] = arr[i + 1];
                arr[i + 1] = k;
                v = 1;
            }
        }
        if (!v) break;
        v = 0;
        end--;
        for (int i = end - 1; i >= str; --i) {
            if (arr[i] > arr[i + 1]) {
                int k = arr[i];
                arr[i] = arr[i + 1];
                arr[i + 1] = k;
                v = 1;
            }
        }
        str++;
    }
}
